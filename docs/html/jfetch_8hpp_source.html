129 bool perform_request(std::string& response) const {
130 std::unique_ptr<CURL, decltype(&curl_easy_cleanup)> curl(curl_easy_init(), curl_easy_cleanup);
131 if (!curl) {
132 throw JFetchException("Failed to initialize CURL");
133 }
134
135 curl_easy_setopt(curl.get(), CURLOPT_URL, url_.c_str());
136 curl_easy_setopt(curl.get(), CURLOPT_CUSTOMREQUEST, method_to_string().c_str());
137 curl_easy_setopt(curl.get(), CURLOPT_WRITEFUNCTION, write_callback);
138 curl_easy_setopt(curl.get(), CURLOPT_WRITEDATA, &response);
139 curl_easy_setopt(curl.get(), CURLOPT_FOLLOWLOCATION, 1L);
140 curl_easy_setopt(curl.get(), CURLOPT_TIMEOUT, 10L);
141
142 struct curl_slist* header_list = nullptr;
143 for (const auto& header : headers_) {
144 header_list = curl_slist_append(header_list, header.c_str());
145 }
146 curl_easy_setopt(curl.get(), CURLOPT_HTTPHEADER, header_list);
147
148 if (!body_.empty() &&
149 (method_ == RequestMethod::POST ||
150 method_ == RequestMethod::PUT ||
151 method_ == RequestMethod::PATCH)) {
152 curl_easy_setopt(curl.get(), CURLOPT_POSTFIELDS, body_.c_str());
153 }
154
155 CURLcode res = curl_easy_perform(curl.get());
156 long http_status;
157 curl_easy_getinfo(curl.get(), CURLINFO_RESPONSE_CODE, &http_status);
158 curl_slist_free_all(header_list);
159
160 if (res != CURLE_OK) {
161 throw JFetchHTTPException(http_status, curl_easy_strerror(res));
162 }
163
164 if (http_status < 200 || http_status >= 300) {
165 throw JFetchHTTPException(http_status, "Unexpected HTTP status code");
166 }
167
168 return true;
169 }
170
171private:
172 std::string url_;
173 RequestMethod method_;
174 std::vector<std::string> headers_;
175 std::string body_;
176
180 std::string method_to_string() const {
181 switch (method_) {
182 case RequestMethod::GET: return "GET";
183 case RequestMethod::POST: return "POST";
184 case RequestMethod::PUT: return "PUT";
185 case RequestMethod::DEL: return "DELETE";
186 case RequestMethod::PATCH: return "PATCH";
187 default: return "GET";
188 }
189 }

194 static size_t write_callback(void* contents, size_t size, size_t nmemb, std::string* user_data) {
195 user_data->append(static_cast<char*>(contents), size * nmemb);
196 return size * nmemb;
197 }

226 T fetch(const std::string& endpoint,
227 const std::unordered_map<std::string, std::string>& query_params = {},
228 const std::vector<std::string>& custom_headers = {},
229 const std::string& custom_body = "") {
230 std::string raw_json;
231
232 // construct the full URL with query parameters
233 std::string full_url = get_base() + endpoint;
234 if (!query_params.empty()) {
235 full_url += "?";
236 for (const auto& [key, value] : query_params) {
237 full_url += key + "=" + value + "&";
238 }
239 full_url.pop_back(); // remove the trailing '&'
240 }
241
242 // combine global and custom headers
243 std::vector<std::string> headers = global_headers;
244 headers.insert(headers.end(), custom_headers.begin(), custom_headers.end());
245
246 // use the provided body, otherwise fallback to the default body
247 std::string body = custom_body.empty() ? get_body() : custom_body;
248
249 // get request method from the endpoint lookup table
250 if (endpoint_lookup.find(endpoint) == endpoint_lookup.end()) {
251 throw JFetchEndpointNotFoundException(endpoint);
252 }
253
254 const auto& [method, lambda] = endpoint_lookup.at(endpoint);
255
256 HttpClient client(full_url, method, headers, body);
257 client.perform_request(raw_json);
258
259 auto json_data = nlohmann::json::parse(raw_json);
260 return lambda(json_data);
261 }
262
267 void set_global_headers(const std::vector<std::string>& headers) {
268 global_headers = headers;
269 }

271protected:
276 virtual std::string get_base() const = 0;
281 virtual std::string get_body() const { return ""; }
